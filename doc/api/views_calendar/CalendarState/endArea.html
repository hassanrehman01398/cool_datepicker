    sizeFactor: betweenControllerList[index],

    child: Align(
      alignment: Alignment.centerLeft,
      child: Container(
        width: width / 14,
        height: width / 10,

          boxShadow: [
            BoxShadow(

              spreadRadius: 1,

      ),
    ),
  );
